防火墙配置

yum -y install firewall-config //安装防火墙图形化管理
firewall-config //图形化防火墙管理
firewall-cmd –list-all //查看防火墙详细信息
firewall-cmd –reload //重启防火墙
firewall-cmd –add-service=http –permanent //永久设置（否者开机放行就没了） 设置后需重启防火墙
一般不要轻易重启防火墙 //仅限制于实验环节使用
firewall-cmd –remove-service=ssh //删除ssh服务放行
firewall-cmd –remove-service=http –permanent //永久删除ssh服务放行

一般放行防火墙示例
firewall-cmd –add-service=ssh
firewall-cmd –add-service=ssh –permanent
